# Cloudformation

env AWS_PROFILE=chas-aws aws cloudformation create-stack \
 --capabilities CAPABILITY_NAMED_IAM \
 --template-body file://serverless-deploy-user-cf.yml \
 --stack-name serverless-deploy-openrecord-staging \
 --parameters ParameterKey=UserName,ParameterValue=serverless-deploy-openrecord ParameterKey=Service,ParameterValue=openrecord-turntable ParameterKey=Stage,ParameterValue=staging \

env AWS_PROFILE=serverless-deploy-openrecord npx sls deploy -s staging
